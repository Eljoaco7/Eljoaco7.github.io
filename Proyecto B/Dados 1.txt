import random
tiradas = int(input("Cuantas tiradas deseas hacer?"))
total = 0
contador = 0 
while(contador<tiradas):
 dado1 = random.randint(1 ,6)
 dado2 = random.randint(1, 6)
 sumadelatirada = dado1 + dado2
 total = total +sumadelatirada
 contador += 1
 print("Tirada", contador, ": dado1 =", dado1, ", dado2 =", dado2, ", suma =", sumadelatirada)
print ("Suma total" ,total,)
